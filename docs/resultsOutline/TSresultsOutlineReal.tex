\section*{Research questions}
% <><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><>
\begin{enumerate}
\item Does tree growth increase with longer growing seasons?
\item Did an earlier start and later end of season increase growth? 
\item Does tree growth increase with longer growing seasons and is it consistent across species?
\item How did growth change across years, and how did it relate to climate?
\end{enumerate}

% <><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><>
\section*{Results}
% <><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><><>
\begin{enumerate}
\item Summary basic information
  \begin{enumerate}
  \item Coringtreespotters includes eleven with leaf phenology data spaning nine years (2016 to 2024) for a total of 
  326 leafout and 
  289 colored leaves observations, yielding 
  278 complete growing seasons (GS) for 
  49 individuals. 
  
  \item 
  2017 was the coolest year
  (21.53$^\circ$C), and   
  2020 was the warmest
  (23.07$^\circ$C)
  
  \item 2020 and 2022 had moderate droughts that ramped up to severe droughts in September. 2016 had a moderate drought starting in June, ramped up to severe in July, then to extreme in August and September. There were no droughts in the other years.
  
  \item GSL ranged from 
  77 days ({\Afla} in 
  2018) to 
  194 days ({\Asac} in 
  2021)

  \item GS thermal sum ranged from 
  1048.8 GDD ({\Afla} in 
  2020) to 
  2634.7 GDD ({\Tame} in 
  2016)
  
  \item The earliest leafout day of year was 
  08-Apr ({\Afla} in 
  2021) and the latest, 
  15-May ({\Cova} in 
  2020). The earliest colored leaves day of year was
  15-Jul ({\Afla} in 
  2021---same individual with earliest leafout) and the latest, 
  06-Nov ({\Tame} in
  2016) 
  
  \item Annual ring widths across years varied from 
  0.43 mm ({\Asac} in 2019) to 
  11.75 mm ({\Tame} in 2019) (Figure \ref{fig:tsboxplot}). 
 
\end{enumerate}

% Question 1 --- --- --- --- --- --- --- --- --- --- --- --- --- --- --- --- ---
\item Does tree growth increase with longer growing seasons?
  \begin{enumerate}

  \item No clear trend of growing season length effect across our eleven species and season predictors.

  \item Longer thermal season increased growth for no species and decreased growth for {\Ball} ($\beta = $
  -0.23, $UI_{90}$
  [-0.36,
  -0.1])
  ) (Figure \ref{fig:tsmuplotbspp}a and e) (Table \ref{tab:ts_slope_table}).
  
  \item No species responded to longer calendar season (Figure \ref{fig:tsmuplotbspp}b and f) (Table \ref{tab:ts_slope_table}).

  \end{enumerate}
  
% Question 2 --- --- --- --- --- --- --- --- --- --- --- --- --- --- --- --- ---
\item Did a later start and end of season increase growth?
  \begin{enumerate}
  \item An earlier start of season increased growth for no species and decreased growth for {\Tame} ($\beta = $
  0.15, $UI_{90}$
  [0.03,
  0.27]) (Figure \ref{fig:tsmuplotbspp}c and g) (Table \ref{tab:ts_slope_table}). 

\item A later end of season increased growth for no species and decreased growth for {\Ball} ($\beta = $
-0.09, $UI_{90}$[-0.16,
-0.01])
(Figure \ref{fig:tsmuplotbspp}d and h) (Table \ref{tab:ts_slope_table}). 

\item The estimates were unchanged when using the full or restricted datasets (SOS $n =$
326 and EOS $n =$
289 vs $n=$
278 Figure \ref{fig:tsSOSEOScomparison}b), or when using budburst instead of leafout as the start of the growing season (Figure \ref{fig:buburstVSleafoutTS}c).

\end{enumerate}

% Question 3 --- --- --- --- --- --- --- --- --- --- --- --- --- --- --- --- ---
\item How did tree growth change across years and tree ids?
  \begin{enumerate}
  
  \item We did not find any clear effect of year on growth (Figure \ref{fig:tsmuplotayear}) and there is no clear differences of growth looking across species (Figure \ref{fig:tsxyGDDyr} and Figure \ref{fig:tsxyGSLyr})
  
  \item Growth did not change across any tree id (Figure \ref{fig:tsmuplotatreeid}).

  \end{enumerate}
\end{enumerate}
